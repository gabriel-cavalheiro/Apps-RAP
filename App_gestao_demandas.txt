CLASS zbp_cds_gest_func IMPLEMENTATION.

  METHOD send_email.

    DATA:
      lo_send_request TYPE REF TO cl_bcs,
      lo_document     TYPE REF TO cl_document_bcs,
      lo_sender       TYPE REF TO cl_cam_address_bcs,
      lo_recipient    TYPE REF TO cl_cam_address_bcs,
      lt_recipients   TYPE TABLE OF mailaddress,
      lt_body         TYPE soli_tab,
      lv_subject      TYPE so_obj_des,
      lv_sender       TYPE mailaddress,
      lv_recipient    TYPE mailaddress.

    TRY.
        " Criar objeto de requisição de envio
        lo_send_request = cl_bcs=>create_persistent( ).

        lv_subject = iv_assunto.
        lv_sender = iv_sender.

        " Definir remetente
        lo_sender = cl_cam_address_bcs=>create_internet_address(
                      i_address_string = lv_sender ).
        lo_send_request->set_sender( i_sender = lo_sender ).


        SELECT EmailAddress
        FROM zicds_gest_email
        INTO TABLE @lt_recipients
        WHERE UserEmail = @sy-uname.

        IF iv_tipo_evento = 'CRIAÇÃO'.

          DATA(lv_team_abap_email) = 'abap.coe@votorantim.com'.
          APPEND lv_team_abap_email TO lt_recipients.

        ELSE.

          SELECT SINGLE EmailAddress
          FROM zicds_gest_email
          INTO @DATA(lv_dev)
          WHERE UserEmail = @iv_desenvolvedor.

          IF lv_dev IS NOT INITIAL.
            APPEND lv_dev TO lt_recipients.
          ENDIF.
        ENDIF.

        LOOP AT lt_recipients INTO lv_recipient.
          lo_recipient = cl_cam_address_bcs=>create_internet_address(
                           i_address_string = lv_recipient ).
          lo_send_request->add_recipient(
            EXPORTING
              i_recipient = lo_recipient
              i_express   = abap_true
          ).
        ENDLOOP.

        " Criar corpo do e-mail
        lt_body = create_email_body(
          iv_demandid    = iv_demandid
          iv_titulo      = iv_titulo
          iv_modulo      = iv_modulo
          iv_mudanca     = iv_mudanca
          iv_request     = iv_request
          iv_tipo_demanda = iv_tipo_demanda
          iv_descricao   = iv_descricao
          iv_prioritario = iv_prioritario
          iv_status      = iv_status
          iv_comentario  = iv_comentario
        ).

        " Criar documento
        lo_document = cl_document_bcs=>create_document(
          i_type      = 'HTM'
          i_text      = lt_body
          i_importance = '1'
          i_subject   = lv_subject
        ).

        " Adicionar documento à requisição
        lo_send_request->set_document( lo_document ).

        lo_send_request->send(
          EXPORTING
            i_with_error_screen = abap_false
        ).

      CATCH cx_bcs INTO DATA(lo_bcs_exception).
        RETURN.
    ENDTRY.

  ENDMETHOD.

  METHOD create_email_body.

    DATA:
      lv_body       TYPE string_data,
      lv_line       TYPE soli,
      lv_offset     TYPE i,
      lv_length     TYPE i,
      lv_max_length TYPE i VALUE 255.

    lv_body = |<html><body>|.

    " Cabeçalho do e-mail
    lv_body = |{ lv_body }<div style="background-color:#0066cc; padding:10px; color:white; font-family:Arial;">|.
    lv_body = |{ lv_body }<h1>Gestão de Demandas</h1>|.
    lv_body = |{ lv_body }</div>|.

    " Conteúdo principal
    lv_body = |{ lv_body }<div style="padding:15px; font-family:Arial;">|.
    lv_body = |{ lv_body }<h2 style="color:#0066cc;">{ iv_titulo }</h2>|.

    lv_body = |{ lv_body }<table border="1" cellpadding="8" cellspacing="0" style="border-collapse:collapse; width:100%;">|.
    lv_body = |{ lv_body }<tr bgcolor="#eeeeee"><th colspan="2" align="left">Informações da Demanda</th></tr>|.

    lv_body = |{ lv_body }<tr><td width="30%"><b>Tipo de Demanda:</b></td><td>{ iv_tipo_demanda }</td></tr>|.
    lv_body = |{ lv_body }<tr><td width="30%"><b>Request:</b></td><td>{ iv_request }</td></tr>|.
    lv_body = |{ lv_body }<tr><td><b>Mudança:</b></td><td>{ iv_mudanca }</td></tr>|.

    lv_body = |{ lv_body }<tr><td><b>Título:</b></td><td>{ iv_titulo }</td></tr>|.
    lv_body = |{ lv_body }<tr><td><b>Módulo:</b></td><td>{ iv_modulo }</td></tr>|.

    IF iv_descricao IS NOT INITIAL.
      lv_body = |{ lv_body }<tr><td><b>Descrição:</b></td><td>{ iv_descricao }</td></tr>|.
    ENDIF.

    IF iv_prioritario IS NOT INITIAL.
      DATA(lv_prioridade_texto) = COND #( WHEN iv_prioritario = 'X' THEN '<span style="color:red; font-weight:bold;">Alta</span>' ELSE 'Normal' ).
      lv_body = |{ lv_body }<tr><td><b>Prioridade:</b></td><td>{ lv_prioridade_texto }</td></tr>|.
    ENDIF.

    IF iv_status IS NOT INITIAL.
      lv_body = |{ lv_body }<tr><td><b>Status:</b></td><td>{ iv_status }</td></tr>|.
    ENDIF.

    IF iv_comentario IS NOT INITIAL.
      lv_body = |{ lv_body }<tr><td><b>Comentário:</b></td><td>{ iv_comentario }</td></tr>|.
    ENDIF.

    lv_body = |{ lv_body }</table>|.

    " Botão para acessar a demanda
    lv_body = |{ lv_body }<p style="margin-top:20px;"><a href="https://fiori.empresa.com.br/gestaodemandas?DemandId={ iv_demandid }" style="background-color:#0066cc; color:white; padding:10px; text-decoration:none;">Acessar Demanda</a></p>|.
    lv_body = |{ lv_body }</div>|.

    " Rodapé
    lv_body = |{ lv_body }<div style="background-color:#eeeeee; padding:10px; font-size:12px; color:#666666; font-family:Arial;">|.
    lv_body = |{ lv_body }<p>Este é um e-mail automático do sistema de Gestão de Demandas. Por favor, não responda a este e-mail.</p>|.
    lv_body = |{ lv_body }</div>|.
    lv_body = |{ lv_body }</body></html>|.

    " Converter string para tabela SOLI
    lv_length = strlen( lv_body  ).
    lv_offset = 0.

    WHILE lv_offset < lv_length.
      " Calcular o tamanho da próxima linha
      DATA(lv_line_length) = nmin( val1 = lv_max_length val2 = lv_length - lv_offset ).

      " Extrair a linha
      lv_line = lv_body+lv_offset(lv_line_length).

      APPEND lv_line TO rt_body.

      " Atualizar o offset
      lv_offset = lv_offset + lv_line_length.
    ENDWHILE.

  ENDMETHOD.

ENDCLASS.

  METHOD assignId.

    READ ENTITIES OF zicds_gest_func IN LOCAL MODE
        ENTITY Demanda
        FIELDS ( DemandId )
        WITH CORRESPONDING #( keys )
        RESULT DATA(lt_result).

    LOOP AT lt_result ASSIGNING FIELD-SYMBOL(<fs_result>).

      IF <fs_result>-DemandId IS INITIAL.

        DATA: l_seq      TYPE numc5,
              l_range_nr TYPE inri-nrrangenr VALUE '01'.

        CALL FUNCTION 'NUMBER_GET_NEXT'
          EXPORTING
            object      = 'ZDMND'
            nr_range_nr = l_range_nr
          IMPORTING
            number      = l_seq
          EXCEPTIONS
            OTHERS      = 1.
        ASSERT sy-subrc = 0.

        <fs_result>-DemandId = |{ l_seq WIDTH = 5 PAD = '0' }|.

      ENDIF.
    ENDLOOP.

    MODIFY ENTITIES OF zicds_gest_func IN LOCAL MODE
    ENTITY Demanda
    UPDATE FIELDS ( DemandId )
    WITH VALUE #( FOR ls_result IN lt_result
                (   DemandUuid = ls_result-DemandUuid
                    DemandId = ls_result-DemandId ) )
                    MAPPED DATA(ls_mapped)
                    FAILED DATA(ls_failed)
                    REPORTED DATA(lr_reported).

    reported-demanda = CORRESPONDING #( lr_reported-demanda ).

  ENDMETHOD.
